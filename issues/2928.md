```sh

do
  case "${options}"
  in
  f) FTR="${ARG}"
    \?) exit;;
  esac
done
```

```sh
while getoptions f option
do
  case "${options}"
  in
  f) FTR="${ARG}";;
    \?) exit;;
  esac
done
```
