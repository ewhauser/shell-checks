import React, {useEffect, useRef, useState} from 'react';
import {View, Text, Pressable, FlatList, Alert} from 'react-native';
import BleManager from 'react-native-ble-manager';
import {Buffer} from 'buffer';

const SVC = 'f0000001-0451-4000-b000-000000000000';
const CFG = 'f0001003-0451-4000-b000-000000000000';

const cues = [
  {label: 'HOLD',    type: 1, side: 0, intensity: 3},
  {label: 'SPEED',   type: 2, side: 0, intensity: 3},
  {label: 'SHIFT L', type: 3, side: 1, intensity: 3},
  {label: 'SHIFT R', type: 3, side: 2, intensity: 3},
  {label: 'UNSAFE',  type: 4, side: 0, intensity: 5},
];

export default function TestHaptique({route}: any) {
  const {deviceId} = route.params;
  const [ready, setReady] = useState(false);
  const rssiTimer = useRef<any>(null);
  const [rssi, setRssi] = useState<number | null>(null);

  async function connectWithTimeout(id: string, ms = 15000) {
    return new Promise(async (resolve, reject) => {
      const t = setTimeout(() => reject(new Error('BLE connect timeout')), ms);
      try {
        await BleManager.connect(id);
        clearTimeout(t);
        resolve(true);
      } catch (e) {
        clearTimeout(t);
        reject(e);
      }
    });
  }

  async function prep() {
    try {
      await connectWithTimeout(deviceId, 15000);
      await BleManager.retrieveServices(deviceId);
      try { await BleManager.requestConnectionPriority(deviceId, 1); } catch {}
      try { await BleManager.requestMTU(deviceId, 247); } catch {}
      rssiTimer.current = setInterval(async () => {
        try { const v = await BleManager.readRSSI(deviceId); setRssi(v); } catch {}
      }, 1500);
      setReady(true);
    } catch (e:any) {
      Alert.alert('Connexion BLE', e?.message ?? 'Erreur connexion');
    }
  }

  async function sendCue(c: any) {
    if (!ready) return;
    const payload = Buffer.from([c.type, c.side, c.intensity]);
    await BleManager.write(deviceId, SVC, CFG, Array.from(payload));
  }

  useEffect(() => {
    return () => { if (rssiTimer.current) clearInterval(rssiTimer.current); };
  }, []);

  return (
    <View style={{padding:16, gap:12}}>
      <Pressable onPress={prep} style={{padding:12, borderWidth:1}}>
        <Text>{ready ? 'Ready' : 'Connect & Prep'}</Text>
      </Pressable>
      <Text>RSSI: {rssi ?? '--'} dBm</Text>
      <FlatList
        data={cues}
        keyExtractor={(i)=>i.label}
        renderItem={({item})=>(
          <Pressable onPress={()=>sendCue(item)} style={{padding:12, borderWidth:1, marginBottom:8}}>
            <Text>{item.label}</Text>
          </Pressable>
        )}
      />
    </View>
  );
}

#include <zephyr/kernel.h>
#include <zephyr/logging/log.h>

LOG_MODULE_REGISTER(ghost_gatt, LOG_LEVEL_INF);

#define BT_UUID_GHOST_SVC_VAL BT_UUID_128_ENCODE(0xf0000001, 0x0451, 0x4000, 0xb000, 0x000000000000)
#define BT_UUID_GHOST_CUE_VAL BT_UUID_128_ENCODE(0xf0001002, 0x0451, 0x4000, 0xb000, 0x000000000000)
#define BT_UUID_GHOST_CFG_VAL BT_UUID_128_ENCODE(0xf0001003, 0x0451, 0x4000, 0xb000, 0x000000000000)

static struct bt_uuid_128 SVC_UUID = BT_UUID_INIT_128(BT_UUID_GHOST_SVC_VAL);
static struct bt_uuid_128 CUE_UUID = BT_UUID_INIT_128(BT_UUID_GHOST_CUE_VAL);
static struct bt_uuid_128 CFG_UUID = BT_UUID_INIT_128(BT_UUID_GHOST_CFG_VAL);

static uint8_t cue_buf[7];
static bool cue_ntf_enabled = false;

static void cue_ccc_changed(const struct bt_gatt_attr *attr, uint16_t value)
{
    cue_ntf_enabled = (value == BT_GATT_CCC_NOTIFY);
    LOG_INF("CUE notify %s", cue_ntf_enabled ? "ON" : "OFF");
}

static ssize_t cfg_write(struct bt_conn *conn, const struct bt_gatt_attr *attr,
                         const void *buf, uint16_t len, uint16_t offset, uint8_t flags)
{
    if (offset != 0 || len < 3) return BT_GATT_ERR(BT_ATT_ERR_INVALID_OFFSET);
    const uint8_t *p = buf;
    uint32_t t_ms = (uint32_t)k_uptime_get_32();
    cue_buf[0] = p[0]; cue_buf[1] = p[1]; cue_buf[2] = p[2];
    memcpy(&cue_buf[3], &t_ms, 4);
    if (cue_ntf_enabled) {
        extern struct bt_gatt_service_static ghost_svc;
        /* NULL = broadcast all; conn = specific client */
        bt_gatt_notify(NULL, &ghost_svc.attrs[2], cue_buf, sizeof(cue_buf));
    }
    return len;
}

BT_GATT_SERVICE_DEFINE(ghost_svc,
    BT_GATT_PRIMARY_SERVICE(&SVC_UUID),
    BT_GATT_CHARACTERISTIC(&CUE_UUID.uuid, BT_GATT_CHRC_NOTIFY, BT_GATT_PERM_NONE, NULL, NULL, NULL),
    BT_GATT_CCC(cue_ccc_changed, BT_GATT_PERM_READ | BT_GATT_PERM_WRITE),
    BT_GATT_CHARACTERISTIC(&CFG_UUID.uuid, BT_GATT_CHRC_WRITE, BT_GATT_PERM_WRITE, NULL, cfg_write, NULL)
);

int ghost_gatt_send_cue(uint8_t type, uint8_t side, uint8_t intensity, uint32_t t_ms)
{
    cue_buf[0] = type; cue_buf[1] = side; cue_buf[2] = intensity;
    memcpy(&cue_buf[3], &t_ms, 4);
    if (!cue_ntf_enabled) return -EACCES;
    return bt_gatt_notify(NULL, &ghost_svc.attrs[2], cue_buf, sizeof(cue_buf));
}
