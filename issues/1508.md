else

fi

if ! [[ $rev_last_retag =~ ^[0-9]+$ ]]

echo -e "$(date +%Y-%m-%d-%H:%M:%S)\t${rev_lastmod}" >> ~/.offlineimap/Mail/.notmuch/last_retag
