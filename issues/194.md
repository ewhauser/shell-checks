Hi, see example test expression in this function:

``` bash
function minecraft {
    typeset \
        jvm=${JAVA_HOME}/bin/java \
        worldPath=${HOME}/doc/mcdata2 \
        serverPath=${HOME}/doc/programs/minecraft_server.14w28b.jar \
        clientPath=${HOME}/doc/programs/Minecraft.jar

    [[
        -n $1 &&
        -d $worldPath &&
        -x $jvm
    ]] || return

    case $1 in
        --server|-s)
            pushd "$worldPath" || return
            "$jvm" -Xmx4096M -jar "$serverPath" nogui &
            printf 'Minecraft server PID: %d\n' "$!"
            fg
            popd
            ;;
        --client|-c)
            "$jvm" -jar "$clientPath"
            ;;
        *)
            echo 'minecraft [ --client | --server ]' >&2
            return 1
    esac
}
```

Interestingly `(())` works fine across multiple lines however.
