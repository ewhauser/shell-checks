This would enable syntax highlighting in the code, and is widely accepted by popular markdown dialects including GFM and krandown.

Run this snippet to deal with it:

`````` Bash
# bash in a naive sence
issue462(){
  sed -i -e s'/\r\n/\n/'g -- "$@" # CRLF screws git up and I am too lazy to fix them bashly
  for i; do
    BUF=() toclose=0
    mapfile -t BUF < "$i" # local IFS=''; while read -r t; do BUF+=("$t"); done < "$i"
    for ((n=0; n<"${#BUF[@]}"; n++)); do
      case "${BUF[n]}" in
      '    '*)
        if [[ "${BUF[n-1]}" == '' ]]; then echo '```sh'; toclose=1; fi
        echo "${BUF[n]#    }";;
      '')
        if [[ "${BUF[n-1]}" == '    '* ]]; then echo '```'; toclose=0; fi
        echo;;
      *)
        echo -E "${BUF[n]}";;
      esac
    done > "$i"
    ((toclose)) && echo '```' >> "$i"
  done
}
