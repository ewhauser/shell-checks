case $1 in 
"start")
	for host in hadoop102 hadoop103 hadoop104
	do
			echo ===================start $host executor server===================
			ssh $host cd /opt/module/azkaban/azkaban-exec;bin/start-exec.sh
	done
	
	sleep 3s
	
	for host in hadoop102 hadoop103 hadoop104
	do
			echo ===================active $host executor server===================
			ssh $host cd /opt/module/azkaban/azkaban-exec;curl -G "$host:$(<./executor.port)/executor?action=activate" && echo
	done
	echo ===================start hadoop102 wed server===================
	ssh hadoop102 /opt/module/azkaban/azkaban-web/bin/start-wed.sh
;;
"stop")
	for host in hadoop102 hadoop103 hadoop104
	do
			echo ===================shutdown $host executor server===================
			ssh $host /opt/module/azkaban/azkaban-exec/bin/shutdown-exec.sh
	done
	echo ===================shutdown hadoop102 wed server===================
	ssh hadoop102 /opt/module/azkaban/azkaban-web/bin/shutdown-web.sh
;;
*)
	echo "请传入正确参数 {start|stop}"
;;
esac
