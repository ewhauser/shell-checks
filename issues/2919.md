#### Here's a snippet or screenshot that shows the problem:

(requires bash ≥ 4.3)
```sh
#!/bin/bash

# outvar is a nameref variable that allows us to set the value of a variable 
# outside our scope
demo_bug() {
   local x="$1"
   local -n outvar=$2

   # do things with x
   x="foo$x"

   outvar="$x"
}

declare foo

demo_bug "bar" foo

echo "$foo"
```
