#!/bin/bash
date_now=$(date +%H:%M:%S)
date_ago=$(date -d"5 minute ago" +%H:%M:%S)
RS_NEW=$(date +%d)/$(date +%b)/$(date +%Y:)
success=$(awk 'BEGIN{RS="'$RS_NEW'"} $1>"'$date_ago'" && $1<"'$date_now'" {print $0}' /XXX/log | grep -w "$1" | grep -cw 200)
total=$(awk 'BEGIN{RS="'$RS_NEW'"} $1>"'$date_ago'" && $1<"'$date_now'" {print $0}' /XXX/log | grep -cw "$1")
if [ "$total" -gt 0 ];then
    echo "scale=2; ($success / $total) * 100" | bc
else
    echo "100"
fi

#### <img src="https://avatars.githubusercontent.com/u/2199511?v=4" width="50">[Nightfirecat](https://github.com/Nightfirecat) commented at [2018-10-30T05:36:27Z](https://github.com/koalaman/shellcheck/issues/1377#issuecomment-434177995):

This is not the right place to go for script  help, but you should fix lines 8 and 9 to read like this:

```
success=$(awk 'BEGIN{RS="'"$RS_NEW"'"} $1>"'"$date_ago"'" && $1<"'"$date_now"'" {print $0}' /XXX/log | grep -w "$1" | grep -cw 200)
total=$(awk 'BEGIN{RS="'"$RS_NEW"'"} $1>"'"$date_ago"'" && $1<"'"$date_now"'" {print $0}' /XXX/log | grep -cw "$1")
```

That is, adding double quotes `"` around all of the BASH variables in those commands.
