function install_mq()
{
  ansible all -m shell -a  'docker load -i /root/rabbitmq-3.6.5.tar'
  sh /root/h3cloud-base/scripts/prepare_rabbitmq_cluster.sh
  sh /root/h3cloud-base/scripts/post_rabbitmq_cluster.sh
  sleep 5
  for i in `kubectl get pod |grep rabbit|awk '{print $1}'`
    do
      kubectl exec -it $i rabbitmq-plugins enable rabbitmq_management
      echo $i
    done
}

function install_redis()

 bash /root/h3cloud-base/scripts/redisClusterPreCmd.sh
 bash h3cloud-base/scripts/redisClusterPostCmd.sh
 kubectl apply -f /root/h3cloud-base/yaml/confFile-cluster/redis-cluster-rc.yaml
 kubectl apply -f /root/h3cloud-base/yaml/confFile-cluster/redis-cluster-service.yaml
}
