#### Here's a snippet or screenshot that shows the problem:

```sh

#!/bin/bash

declare -a map_a=(1 2 3)
declare -a map_b=(a b c)

for var in ${!map_@}; do
    eval keys=(\${!$var[@]})
    eval values=(\${$var[@]})

    echo "${keys[@]}"
    echo "${values[@]}"
done
```
