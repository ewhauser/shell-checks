#### Here's a snippet or screenshot that shows the problem:

Script snapshot is called with two possibilities:
`script_name.sh "unifi"`
`script_name.sh "streaming"`

Script will check host_name=$1 for any configured apps and their paths/schedule to backup

```
#!/bin/bash

target="/mnt/pool/data/backup"

minidlna_paths=(
    "/usr/local/etc/minidlna.conf"  "always"
    "/usr/local/etc/rc.d/minidlna"  "always"
)

plex_paths=(
    "/usr/local/etc/plexmedia.conf"     "Mon"
    "/data/plex_catalog"                "Monday"
)

unifi_controller_paths=(
    "/usr/local/share/java/unifi/data"          "Sun"
    "/usr/local/share/java/unifi/data/backup"   "always"
)

unifi_ap_paths=(
    "/data/local/unifi_ap"   "Sun"      
    "/share/unifi_ap/data"   "always"
)

declare -A allAppPaths
allAppPaths[minidlna]="minidlna_paths[@]"
allAppPaths[plex]="plex_paths[@]"
allAppPaths[unifi_controller]="unifi_controller_paths[@]"
allAppPaths[unifi_ap]="unifi_ap_paths[@]"

# streaming host has minidlna app to backup
allAppsIn_streaming=(
    "minidlna"
    "plex"
)

# unifi host has unifi controller app to backup
allAppsIn_unifi=(
    "unifi_controller"
)

declare -A allAppsInHost
allAppsInHost[streaming]="allAppsIn_streaming[@]"
allAppsInHost[unifi]="allAppsIn_unifi[@]"

hostname="$1"
apps_in_host=("${!allAppsInHost[$hostname]}") # array with all app names in $host_name script arg

echo "> Backup apps and config files..."
echo "...found ${#apps_in_host[@]} apps(s) to backup in host=$hostname"

i=1
for appname in "${apps_in_host[@]}"; do
    app_paths=("${!allAppPaths[$appname]}") # array with all path/schedule pairs for the app $appname
    dest_dir="$target/$hostname/$appname"

    echo ""
    echo "$i". "$appname"
    ((i++))

    # Create app target directory
    echo ""
    echo "   > creating target directory:"
    echo "     $appname --> $dest_dir"
    echo ""

    # Check source app paths to backup if they exist or are empty directories
    echo "   > checking app source paths to backup:"
    for ((j=0; j<${#app_paths[@]}; j=$((j+2)))); do
        app_path_url="${app_paths[j]}"
        sched_index=$((j+1))
        app_path_schedule="${app_paths[$sched_index]}"

        # check if schedule allows backup to be done for the current path
        schedule="false"
        [ "$app_path_schedule" = "always" ] && schedule="true"

        # - day of the week short format: Mon,Tue,Wed,Thu,Fri,Sat,Sun
        sched_day=$(date +'%a')
        [ "$app_path_schedule" = "$sched_day" ] && schedule="true"

        # day of the week long format: Monday,Tuesdaye,Wednesday,Thursday,Friday,Saturday,Sunday
        sched_day=$(date +'%A')
        [ "$app_path_schedule" = "$sched_day" ] && schedule="true"

        # day number of the month, single digits are preceded by a blank !
        # - remove leading spaces
        sched_day=$(date +'%e')
        sched_day="${sched_day#"${sched_day%%[![:space:]]*}"}"
        [ "$app_path_schedule" = "$sched_day" ] && schedule="true"

        # if it is not a schedule day, skip backup
        if [ "$schedule" != "true" ]; then
            echo "     - path $((j/2 + 1))=$app_path_url"
            echo "       > skipped: schedule=$app_path_schedule"
            continue
        fi

        echo "     - path $((j/2 + 1))=$app_path_url"
    done
done
```

#### Here's what shellcheck currently says:
`SC2034 (warning): minidlna_paths appears unused. Verify use (or export if used externally).`


#### Here's what I wanted or expected to see:
Properly detect using associative arrays in bash and not only index arrays to simulate 3D arrays
