#!/bin/bash

echo
echo small-alpha
alpha='ɑ' # latin small letter alpha

echo -n "$alpha" | xxd -p | xxd -p -r # result ɑ
echo -en '\xC9\x91' # result ɑ # latin small letter alpha
printf '\xC9\x91' # result ɑ # latin small letter alpha
echo
printf -v alpha_4xchar '%04x' "'$alpha"
echo $alpha_4xchar # result Unicode 0251
printf -v alpha_2xchar '%02x' "'$alpha"
echo $alpha_2xchar # result Unicode 251 <= why 251 but format is %02x
printf '%02x' "'$alpha"
/bin/printf
