Oh, I do use this snippet in my personal `bl.sh` (bash library)

```shell
# Bash creates some shell vars after the first pipeline is performed
# shellcheck disable=2312
bl::_noop() {
  : | :
}

bl::_before() {
  before_globals=$(mktemp)
  after_globals=$(mktemp)
  new_globals=$(mktemp)
  compgen -A variable > "${before_globals}"
  trap bl::_after EXIT
}

bl::_after() {
  compgen -A variable > "${after_globals}"
  diff -u "${before_globals}" "${after_globals}" > "${new_globals}" || {
    echo 'Leaked global variables:'
    cat "${new_globals}"
  }
}

bl::main() {
  bl::_noop
  bl::_before
  main "$@"
}
```
