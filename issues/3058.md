#### Here's a snippet or screenshot that shows the problem:

```sh
#!/bin/sh
LF='
'

find "$SOURCE" \
    -name '.DS_Store' -o -name 'Thumbs.db' -prune -o \
    -type f \
    -print |
    sort |
    xargs -n "$FILE_LIMIT" |
    while IFS= read -r "-d${LF}" fs; do
        TARGET="${PREFIX}-${I}"
        mkdir -p "$TARGET"

        for f in $fs; do
            cp "$f" "$TARGET"
        done

        I=$((I+1))
    done
```
