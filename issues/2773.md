#### Here's a snippet or screenshot that shows the problem:

```sh
#!/bin/sh

# these cause SC2016 false positives:
command -p -- awk -- '! seen[$0]++ {print}' "$@"
command -p -- awk -- '!seen[$0]++ {print}' "$@"
command -p -- awk '! seen[$0]++ {print}' "$@"
command -p -- awk '!seen[$0]++ {print}' "$@"
command -p awk -- '! seen[$0]++ {print}' "$@"
command -p awk -- '!seen[$0]++ {print}' "$@"
command -p awk '! seen[$0]++ {print}' "$@"
command -p awk '!seen[$0]++ {print}' "$@"

# these pass shellcheck:
command awk -- '! seen[$0]++ {print}' "$@"
command awk '! seen[$0]++ {print}' "$@"
command awk '!seen[$0]++ {print}' "$@"
awk -- '! seen[$0]++ {print}' "$@"
awk '! seen[$0]++ {print}' "$@"
awk '!seen[$0]++ {print}' "$@"
```
