#### Here's a snippet or screenshot that shows the problem:

```sh
case "$SUBCOMMAND" in
begin)
    export CRASH_STATUS='0'
    ;;
do)
    if [ "$#" -lt 2 ]; then
        usage_crash
        return 1
    fi

    shift
    "$@" || export CRASH_STATUS=1
    ;;
end)
    if [ "$CRASH_STATUS" = '1' ]; then
        return 1
    fi
    ;;
help)
    usage_crash
    ;;
*)
    usage_crash
    return 1
    ;;
esac

```
