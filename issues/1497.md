Shellcheck somehow misses the scope of a variable, and figuring out all outstanding SC2034 can be a uninteresting, repetitive task. Follow along:

#### Here's a snippet or screenshot that shows the problem:

```sh
#!/bin/bash

_comp1() {
	local cur opts
	local cur prev opts
	COMPREPLY=()
	cur="${COMP_WORDS[COMP_CWORD]}"
	prev="${COMP_WORDS[COMP_CWORD-1]}"
	opts="a b c"

	mapfile -t COMPREPLY <<< "$(compgen -W "${opts}" -- "${cur}")"
	return 0
}
complete -F _comp1 comp1
```
