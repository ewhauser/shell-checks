#!/usr/bin/env bash

function installFormula()
{
    if [[ -z "$1" ]]
    then
        local EmptyNameError="no name specified"
        local -i EmptyNameErrorCode=1
        echo "$EmptyNameError" >&2
        return $EmptyNameErrorCode
    fi

    name="$1"
    if ! brew list | grep "$name" &> /dev/null
    then
        if ! brew install "$name" &> /dev/null
        then
            local UnknownError="can't install $name, unknown error"
            local -i UnknownErrorCode=1
            echo "$UnknownError" >&2
            return $UnknownErrorCode
        fi
    else
        local AlreadyInstalledError="can't install $name because it is already installed"
        local -i AlreadyInstalledErrorCode=1
        echo "$AlreadyInstalledError" >&2
        return $AlreadyInstalledErrorCode
    fi

    return 0
}
