#### Here's a snippet or screenshot that shows the problem:

```sh
#!/bin/bash

scope_not_used_present()
{
        declare -a foo1
        foo1=(dummy)

        declare -a foo2=(dummy)
}

scope_not_used_missing ()
{
        declare -a foo1a
        foo1a=(dummy)
 
        declare -a foo2a=(dummy)
}

scope_false_override ()
{
        declare foo1a
        foo1a=dummy

        declare foo2a=dummy
}
```
