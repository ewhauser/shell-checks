#
# Check for the presence of a particular character in a script parameter.
#
# @param      haystack   A string to be searched
# @param      needle     A substring to be searched for.
#
# @return:    0 if found.
#
testCharInScriptParameter()
{
    haystack=$1
    needle=$2

    echo "${haystack}" | grep -q "${needle}"
}

# Check for the presence of a '{' character in a parameter used for an alarm level.
# which is an indicator that the alarm is combinatorial.
# Now we should darned wlll know if it is by the value of the "alarm type" field,
#  but we use this as a double-check.
#
# It is conceivable that you would define a combinatorial test as 1 == 2 - so involving no partameters at all.
#  but so unlikely and useless we discount it for now,
#  and anyway this is just double-check code, the first test is the "alarm type" field.
#
# @param      haystack   A string to be searched
#
# @return     As for testCharInScriptParameter
#
testCombinatorial()
{
    haystack=$1
    testCharInScriptParameter "${haystack}" "{"
}

# Check for the presence of a '.' character in a parameter.
# which is an indicator that the value is decimal.
#
# @param      haystack   A string to be searched
#
# @return     As for testCharInScriptParameter
#
testDecimal()
{
    haystack=$1
    testCharInScriptParameter "${haystack}" "."
}

# use sed to remove ALL blanks -- need to ensure that's a valid action -- on alarm expressions for instance.
#                                 probably OK for sms files at present.
#
# @return 0 if a line was obtained.
#
getSmsLine()
{
    # IFS='' (or IFS=) prevents leading/trailing whitespace from being trimmed.
    # -r prevents backslash escapes from being interpreted.
    # testing [ -n $line ] prevents the last line from being ignored if it doesn't end with a \n

    while true 
    do
        IFS='' read -r line; rc_read=$?
        if [ -z "$line" ] && [ "0" -ne "${rc_read}" ]
        then
            # end-of-file AND nothing returned.
            break  # note rc_read is non-zero.
        else
            # There may be SOMETHING in $line, though it can be an empty line from the middle of the file.
            line="$(echo ${line} | grep -v "^$" | grep -v "/" | grep -v "#" | sed 's/ //g' )"
            if [ -n "$line" ]
            then
                # there is SOMETHING 'of significance' in $line.
                rc_read=0 # force OK reply if there is anything to process.
                (( smsFileLine++ ))
                break
            fi
        fi
        # We read a line, but it evaporated to nothing when we grepped it - try for another one.
    done < ${smsInterFile}

    return ${rc_read}
}

        missingLines=true
        break
    fi

    thisSmsNumber="${line}"

    if ! getSmsLine

        missingLines=true
        break

    #    _ _ _     -> space

    #    _ _ s     -> /
    #
    thisSmsMessage="$(echo ${line} | sed 's/___/ /g' | sed 's/__h/#/g' | sed 's&__s&/&g' )"

                    fi

                       [ "+" = "${firstCharOfField}" ] ||
                       [ "/" = "${firstCharOfField}" ] ||

                        #  or we have already processed a * or /

                        alterScaled=true
                    else

                        # We could implement this just as for the other operators,
                        #  but it seems more natural to express a scaling of /100 as
                        #   0 .. 100 -> 0.0 .. 1.0
                        # It's just a nicety we could live without.

                        # We implement an initial divide term by
                        #     altering the unscaled values if the value is integral.
                        #     altering the scaled   values if the value is non-integral.
                        # That avoids the appearance of negative or fractional numbers in the raw values.
                        #
                        #   So an initial
                        #      /7    gives us 0 .. 7 -> 0.0 .. 1.0
                        #      /2.5  gives us 0 .. 1 -> 0.0 .. 0.4
                        if testDecimal "${field}"
                        then
                            alterScaled=true
                        else
                            alterScaled=false

                            #  and multiplying those is equivalent to divif=ding the result.
                            ${field}=${field/\//*/}
                        fi
                    fi

                    #
                    # In a properly-written script we would at this point validate the rest of the filed is a number
                    #  but we simply trust that it is.
                    # Note that leaves us open to 'maths injection' where a field actuually contains '* 27 + z^2'
                    #

                    if ${alterScaled}
                    then
                        # To alter the number of decimal points ('scale') of a number you need to divide - +/-/* won't do it.

                        updateDecimalVariable thisParameterHighScaled  "${field}"
                    else
                        # We have already altered the operator so that changing the raw values does what we need.
                        updateIntegerVariable thisParameterLowRaw   "${field}"
                        updateIntegerVariable thisParameterHighRaw  "${field}"
                    fi
                fi
            done
        else

            thisParameterLowScaled=${scalingFields[1]}
            thisParameterLowRaw=${scalingFields[2]}
            thisParameterHighScaled=${scalingFields[3]}
            thisParameterHighRaw=${scalingFields[4]}
        fi

#### <img src="https://avatars.githubusercontent.com/u/1693206?v=4" width="50">[knirch](https://github.com/knirch) commented at [2016-02-15T12:48:49Z](https://github.com/koalaman/shellcheck/issues/600#issuecomment-184192238):

Shellcheck has no way of telling if a missing " is intentional or not.

``` bash
f="/tmp/testfile"
touch "$f # Create a "test" file
echo "all done"
```

A human can probably determine that there's a missing " on the second line, but the code will try and follow the parsing rules in bash.

``` bash
echo "Example text
segment with
newlines."
```

Counting the number of quotes would be bad too

``` bash
echo "something something" # Wish I had a 20" monitor.
```

IMHO the best way to detect errors like a missing quote is to use an editor which does syntax highlighting.
